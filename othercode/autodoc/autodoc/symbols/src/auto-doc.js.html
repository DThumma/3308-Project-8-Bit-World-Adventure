  7 /** setCookie function 
  8 *sets unique cookie ID for a user when the user types in his username.
  9 */
 10 function setCookie(cname, cvalue, exdays) {
 11     var d = new Date();
 12     d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
 13     var expires = "expires=" + d.toGMTString();
 14     document.cookie = cname + "=" + cvalue + "; " + expires;
 15 }
 16 
 17 /** getCookie function calls saved cookie for a username
 18 *@return saved unique cookie of a username
 19 */
 20 function getCookie(cname) {
 21     var name = cname + "=";
 22     var ca = document.cookie.split(';');
 23     for (var i = 0; i < ca.length; i++) {
 24         var c = ca[i];
 25         while (c.charAt(0) == ' ') c = c.substring(1);
 26         if (c.indexOf(name) == 0) {
 27             return c.substring(name.length, c.length);
 28         }
 29     }
 30     return "";
 31 }
 32 
 33 /** checkCookie function 
 34 *checks the username to differentiate between returning user and new user
 35 *@return setCookie or welcome again text
 36 */
 37 function checkCookie() {
 38     var user = getCookie("username");
 39     if (user != "") {
 40         alert("Welcome again " + user);
 41     } else {
 42         user = prompt("Please enter your name:", "");
 43         if (user != "" && user != null) {
 44             setCookie("username", user, 30);
 45         }
 46     }
 47 }

217 /** -(x){ function 
218     *@return element document using the ID input x, used to put javascript variables into the actial HTML
219     */
220 function _(x) {
221     return document.getElementById(x);
222 }
223 /** renderQuestion function 
224     *prints a question from the question bank saved in questions array
225     *@return question
226     */
227 function renderQuestion() {
228     test = _("test");
229     if (pos >= questions.length) {
230         test.innerHTML = "<h2>You got " + correct + " of " + questions.length + " questions correct</h2>";
231         _("test_status").innerHTML = "Quiz Completed";
232         pos = 0;
233         correct = 0;
234         return false;
235     }
236     _("test_status").innerHTML = "Australia Quiz Question " + (pos + 1) + " of " + questions.length;
237     question = questions[pos][0];  //target the question
238     chA = questions[pos][1];  //target first option
239     chB = questions[pos][2]; //target second option
240     chC = questions[pos][3]; //target first option
241     test.innerHTML = "<h3>" + question + "</h3>";
242     test.innerHTML += "<input type='radio' name='choices' value='A'> " + chA + "<br>";
243     test.innerHTML += "<input type='radio' name='choices' value='B'> " + chB + "<br>";
244     test.innerHTML += "<input type='radio' name='choices' value='C'> " + chC + "<br><br>";
245     test.innerHTML += "<button onclick='checkAnswer()'>Submit Answer</button>";
246 }
247 /** checkAnswer function 
248     *checks for correct answer for each question and prints out correct output of the chosen answer is wrong
249     *@return true or false
250     */
251 function checkAnswer() {
252     choices = document.getElementsByName("choices");
253     for (var i = 0; i < choices.length; i++) {
254         if (choices[i].checked) {
255             choice = choices[i].value;
256         }
257     }
258     if (choice == questions[pos][4]) {
259         correct++;
260         document.getElementById("answer").innerHTML = "CORRECT!"
261     }
262     else {
263         if (questions[pos][4] == 'A') {
264             document.getElementById("answer").innerHTML = "Sorry, the the correct answer to '" + questions[pos][0] + "' was " + questions[pos][1] + "!"
265             //window.alert(questions[pos][4])
266         }
267         if (questions[pos][4] == 'B') {
268             document.getElementById("answer").innerHTML = "Sorry, the the correct answer to '" + questions[pos][0] + "' was " + questions[pos][2] + "!"
269             //window.alert(questions[pos][4])
270         }
271         if (questions[pos][4] == 'C') {
272             document.getElementById("answer").innerHTML = "Sorry, the the correct answer to '" + questions[pos][0] + "' was " + questions[pos][3] + "!"
273             //window.alert(questions[pos][4])
274         }
275     }
276 
277     pos++;
278     renderQuestion();
279 }
280 window.addEventListener("load", renderQuestion, false);
